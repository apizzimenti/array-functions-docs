/**
 * @author Anthony Pizzimenti
 *
 * @desc A singly linked list.
 *
 * @property _head {Nod} The Node object at the head of the list.
 * @property _length {Number} The number of linked Node objects.
 * @property _currentNode {Node} The current referenced Node.
 *
 * @class {object} A singly linked list.
 * @this sList
 * @constructor
 */

function sList () {
    
    this._head = null;
    this._length = 0;
    this._currentNode = null;
}

/**
 * @author Anthony Pizzimenti
 *
 * @param value {*} Value to be assigned to the new Node.
 *
 * @returns {Node}
 *
 * @this sList
 */

sList.prototype.addToTail = function (value) {
    
    var node = new Node(value),
        currentNode = this._head;
    
    if (!currentNode) {
        this._head = node;
        this._length++;
        return node;
        
    }
    
    while (currentNode.next) {
        currentNode = currentNode.next;
    }
    
    currentNode.next = node;
    this._length++;
    return node;
};

/**
 * @author Anthony Pizzimenti
 *
 * @param value {*} Value to be assigned to the new Node.
 *
 * @returns {Node}
 *
 * @this sList
 */

sList.prototype.addToHead = function (value) {
    
    var node = new Node(value),
        currentNode = this._head;
    
    if (!currentNode) {
        this._head = node;
        this._length++;
        return node;
    }
    
    node.next = currentNode;
    this._head = node;
    
    while (currentNode.next) {
        currentNode = currentNode.next;
    }
    
    this._length++;
    return node;
};

/**
 * @author Anthony Pizzimenti
 *
 * @desc Adds a Node with value <code>value</code> at the position <code>position</code>.
 *
 * @param value {*} Value to be assigned to the new Node.
 * @param position {Number} Position in the linked list.
 *
 * @returns {Node}
 *
 * @this sList
 */

sList.prototype.add = function (value, position) {
    
    if (position > this._length) {
        throw new Error("Index out of range.");
    } else if (position === 0) {
        return this.addToHead(value);
    } else if (position === this._length - 1) {
        return this.addToTail(value);
    }
    
    var node = new Node(value),
        currentNode = this._head,
        lastNode,
        nextNode,
        i = 0;
    
    while (true) {
        
        if (i === position - 1) {
            lastNode = currentNode;
        } else if (i === position) {
            nextNode = currentNode;
            break;
        }
        
        i++;
        currentNode = currentNode.next;
    }
    
    this._length++;
    
    lastNode.next = node;
    node.next = nextNode;
    return node;
};

/**
 * @author Anthony Pizzimenti
 *
 * @desc Retrieves the Node at the specified positon.
 *
 * @param position {Number} Position to retrieve Node from.
 *
 * @this sList
 *
 * @returns Node
 */

sList.prototype.retrieve = function (position) {
    
    var currentNode = this._head,
        i = 0;
    
    if (position > this._length) {
        throw new Error("Index out of range.");
    }
    
    while (i < position) {
        currentNode = currentNode.next;
        i++;
    }
    
    return currentNode;
};

/**
 * @author Anthony Pizzimenti
 *
 * @desc Removes the Node from the specified position.
 *
 * @param position {Number} Position from which the Node will be removed.
 *
 * @this sList
 */

sList.prototype.remove = function (position) {
    
    var currentNode = this._head,
        remNode,
        lastNode,
        nextNode,
        i = 0;
    
    if (position > this._length) {
        throw new Error("Index out of range.");
    }
    
    if (position !== 0 || position !== this._length - 1) {
    
        while (i < this._length) {
        
            if (i === position - 1) {
                lastNode = currentNode;
            } else if (i === position) {
                remNode = currentNode;
            } else if (i === position + 1) {
                nextNode = currentNode;
                break;
            }
        
            currentNode = currentNode.next;
            i++;
        }
    
        remNode = null;
        lastNode.next = nextNode;
        
    } else if (position === 0) {
        
        this._head = currentNode.next;
        currentNode = null;
        
    } else if (position === this._length - 1) {
        
        while (i < this._length) {
            
            currentNode = currentNode.next;
            
            if (i === position) {
                currentNode.next = null;
            }
            
            i++;
        }
    }
    
    this._length--;
};
